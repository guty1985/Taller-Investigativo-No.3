import matplotlib.pyplot as plt

x=np.linspace(0,100)
a=8
b=50
c=90

plt.xlabel('Valores de X')
plt.ylabel('Valores de Y')
plt.title('Funcion Triangular')

plt.grid()

def f(x,a,b,c):
    if ((x<a) | (x>=c)):
        ans=0
    if ((a<=x) & (x<b)):
         ans=(x-a)
    if ((b<=x) & (x<=c)):
        ans=(c-x)
    return ans

f_vec = np.vectorize(f)
func=f_vec(x,a,b,c)
plt.plot(x,f_vec(x,a,b,c))

d=70

def f(x,a,b,c,d):
    
    if (x<a)|(x>=c):
        m=0;
   
    if (x>=a)&(x<b):
        m=(x-a)/(b-a)
    
    if (x>=b)&(x<d):
        m=1
    
    if (x>=d)&(x<c):
        m=1-(x-d)/(c-d)
    return m

import numpy as np
import matplotlib.pyplot as plt

x=np.linspace(0,100)

a=10
b=30
c=60

def f(x,a,b,c):
    
   
    if (x<a):
        p=0
  
    if (x>=a)&(x<b):
        p=2*(((x-a)/(c-a))**2)
    
    if (x>=b)&(x<=c):
        p=1-2*(((x-c)/(c-a))**2)
   
    if (x>c):
        p=1

    return p

plt.xlabel('Valores de X')
plt.ylabel('Valores de Y')
plt.title('Funcion Forma S')
